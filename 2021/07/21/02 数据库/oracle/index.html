oracle命令

docker安装

# 拉取项目
docker pull registry.cn-hangzhou.aliyuncs.com/helowin/oracle_11g
# 启动项目
docker run  --privileged --name oracle11g -p 1521:1521  registry.cn-hangzhou.aliyuncs.com/helowin/oracle_11g
# 进入项目
docker exec -it oracle11g /bin/bash

# 切换到root用户,密码为helowin
su - root
vi /etc/profile

输入环境变量到全局

export ORACLE_HOME=/home/oracle/app/oracle/product/11.2.0/dbhome_2
export ORACLE_SID=helowin
export PATH=$ORACLE_HOME/bin:$PATH
# 生效更改的环境变量
source /etc/profile
# 创建软链接
ln -s $ORACLE_HOME/bin/sqlplus /usr/bin
# 创建一个data文件
mkdir /data
chmod -R 777 /data
# 切换到oracle用户
su - oracle

# 使用sysdba登陆
sqlplus / as sysdba
# 无效的话，先登陆再切换用户
sqlplus /nolog

oracle的sql查询和插入会自动转换为大写, 严格匹配要加””号， ‘’号一般表示字符串, “”一般加在表名和列名

分页写法

第一个%s表示要查询的字段, 第二个%s为原本的查询语句, 第三个%s表示order by的限制

SELECT %s FROM (SELECT e.*, ROWNUM rn FROM (%s) e) WHERE ROWNUM <= 10 AND rn > 1 %s

插入多行数据

INSERT ALL
  INTO mytable (column1, column2, column_n) VALUES (expr1, expr2, expr_n)
  INTO mytable (column1, column2, column_n) VALUES (expr1, expr2, expr_n)
  INTO mytable (column1, column2, column_n) VALUES (expr1, expr2, expr_n)
SELECT * FROM dual;
--创建序列
CREATE SEQUENCE sys_log    
    INCREMENT BY 1 -- 每次加几个  
    START WITH 1 -- 迁移过来的，已经有记录了  
    MINVALUE 1      --最小值1  
    NOMAXVALUE -- 不设置最大值   
    NOCYCLE -- 一直累加，不循环  
    NOCACHE -- 不建缓冲区  


CREATE  TRIGGER sys_log_tri    
         BEFORE  
         INSERT ON FX_SYSLOG  
         FOR EACH ROW  
         BEGIN  
         SELECT sys_log.nextval into :new.id  from dual;  
     END sys_log_tri; 

DROP  SEQUENCE sys_log;
DROP  TRIGGER sys_log_tri;
