/*
-d 只下载不安装(通过git clone下载的，没有依赖包)

GRPC

双向流， 发送和接受信息一次大于4mb会报错, 可以通过流来避免

协议

// 返回响应
message StreamMessage {
  bytes  data        =   1;
}

// 定义服务,可定义多个服务,每个服务可多个接口
service DbService {
  // rpc请求 请求的函数 (发送请求参数) returns (返回响应的参数)
  rpc Exec    (stream StreamMessage)    returns(stream StreamMessage);
}

服务端 - 传入

func (s *server) Exec(stream dbs.DbService_ExecServer) error {
    //监听会话是否还有效
    isLive := make(chan bool)
    end := make(chan bool)
    err := make(chan error)
    // 传入流关闭时间,没有设置默认为5s
    sgTimeOut := settings.GConfig.Server.SgTimeOut
    if sgTimeOut == 0 {
        sgTimeOut = 5
    }
    var reqBuffer bytes.Buffer
    // 接受端，协程阻塞等待流获取完毕
    go func() {
        for {
            in, e := stream.Recv()
            if e != nil {    //客户端是go的话，写入结束会返回io.EOF的报错
                err <- e
                return
            }
            reqData := in.GetData()
            if len(reqData) != 3 || string(reqData) != "end" {    //不是go客户端不能自己关闭写入流的话，传输一个标记结束的标记
                reqBuffer.Write(reqData)
                isLive <- true
                continue
            } else {
                end <- true
                return
            }
        }
    }()

    for {
        select {
        case <-isLive:    //判断是不是一直有传输过来
        case <-time.After(time.Duration(sgTimeOut) * time.Second):
            return errors.New("传入会话连续无响应")
        case <-end:
            return doAction(stream, reqBuffer)
        case e := <-err:
            return e
        }
    }

}

服务端 - 返回结果

// 发送大小大于4Mb的分段传输
dataByte := []byte(data)

n := len(dataByte)
// grpc 协议头用了5个字节
byteSize := 4 * 1024 * 1024 - 5
if n < byteSize {
    stream.Send(&dbs.StreamMessage{Data: dataByte})
    return nil
} else {
    for i := 0; i < n; i += byteSize {
        l := i + byteSize
        if l > n {
            l = n
        }
        repData := dataByte[i:l]
        stream.Send(&dbs.StreamMessage{Data: repData})
    }
    return nil
}

交叉编译

linux

#!/bin/bash
export CGO_ENABLED=0    //是否使用cgo
export GOOS=linux        //要编译成的目标系统，默认为当前系统（当前为linux系统可不填）
export GOARCH=amd64        //系统架构, 386，amd64，arm
export GOEXE=.out        //
go build -ldflags "-s -w" -o src.out ./src/main.go //设置编译参数-ldflags "-w -s"。其中-w为去掉调试信息（无法使用gdb调试），-s为去掉符号表
mkdir -p ./bin                //在当前目录创建一个bin目录，创建或替换server执行文件
rm -rvf ./bin/server.out
upx -o ./bin/server.out src.out
rm -rvf src.out

window

#!/bin/bash
export GOOS=windows
export GOARCH=amd64
go build -ldflags "-s -w" ./src/main.go
mkdir -p ./bin
rm -rvf ./bin/server.exe
upx -o ./bin/server.exe main.exe
rm -rvf main.exe
